1. #include <stdio.h>  
  2. #include <stdlib.h>  
  3. //Represent a node of singly linked list  
  4. struct node{  
  5.     int data;  
  6.     struct node *next;  
  7. };      
  8.    
  9. //Represent the head and tail of the singly linked list  
  10. struct node *head, *tail = NULL;  
  11.    
  12. //addNode() will add a new node to the list  
  13. void addNode(int data) {  
  14.     //Create a new node  
  15.     struct node *newNode = (struct node*)malloc(sizeof(struct node));  
  16.     newNode->data = data;  
  17.     newNode->next = NULL;  
  18.       
  19.     //Checks if the list is empty  
  20.     if(head == NULL) {  
  21.         //If list is empty, both head and tail will point to new node  
  22.         head = newNode;  
  23.         tail = newNode;  
  24.     }  
  25.     else {  
  26.         //newNode will be added after tail such that tail's next will point to newNode  
  27.         tail->next = newNode;  
  28.         //newNode will become new tail of the list  
  29.         tail = newNode;  
  30.     }  
  31. }  
  32.    
  33. //display() will display all the nodes present in the list  
  34. void display() {  
  35.     //Node current will point to head  
  36.     struct node *current = head;  
  37.       
  38.     if(head == NULL) {  
  39.         printf("List is empty\n");  
  40.         return;  
  41.     }  
  42.     printf("Nodes of singly linked list: \n");  
  43.     while(current != NULL) {  
  44.         //Prints each node by incrementing pointer  
  45.         printf("%d ", current->data);  
  46.         current = current->next;  
  47.     }  
  48.     printf("\n");  
  49. }  
  50.       
  51. int main()  
  52. {  
  53.     //Add nodes to the list  
  54.     addNode(1);  
  55.     addNode(2);  
  56.     addNode(3);  
  57.     addNode(4);  
  58.       
  59.     //Displays the nodes present in the list  
  60.     display();  
  61.    
  62.     return 0;  
  63. }
